    interpolator = mr.SpiceInterpolator(target_body[0], dates)
    mr.toc()
    mr.tic("Interpolating")
    pts_interp = interpolator(fine_epsecs)
    mr.toc()
    mr.tic("Computing true positions")

    mr.toc()
