    import matplotlib.pyplot as plt
    import numpy as np

    import mirage as mr

    lambdas = np.linspace(350, 800) * 1e-9  # [nm]
    kcd = 1 / mr.candela_to_watt_per_sr(np.ones_like(lambdas), lambdas=lambdas)

    plt.plot(lambdas, kcd)
    mrv.texit(

    )
