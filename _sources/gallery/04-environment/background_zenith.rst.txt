    def plot_sig(pl, s, cmap, scale=1):
        mrv.scatter3(
            pl,
            scale * pts,
            scalars=s,
            cmap=cmap,
            opacity=(s - np.min(s)) / (np.max(s) - np.min(s)),
            point_size=15,
