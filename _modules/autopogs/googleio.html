def validate_email(email_address: str) -> bool:
    import re

    regex = r"\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,7}\b"
    if re.fullmatch(regex, email_address):
        return True
    return False

def update_calendar_event_summary(event_id: str, prepend_msg: str):
    service = get_calendar_service()
    event = (
        service.events()
        .get(calendarId=os.environ["POGS_CALENDAR_ID"], eventId=event_id)
        .execute()
    )
    event["summary"] = f'{prepend_msg}: {event["summary"]}'
    service.events().update(
        calendarId=os.environ["POGS_CALENDAR_ID"],
        eventId=event_id,
        body=event,
        sendUpdates="all",
    ).execute()
    print("Updated event summary")

def get_calendar_events_in_range(
    calendar_id: str, start_time: datetime.datetime, end_time: datetime.datetime
) -> Union[None, List[dict]]:
    service = get_calendar_service()

    # Call the Calendar API
    start_time = start_time.isoformat()
    end_time = end_time.isoformat()
    events_result = (
        service.events()
        .list(
            calendarId=calendar_id,
            timeMin=start_time,
            timeMax=end_time,
            singleEvents=True,
            orderBy="startTime",
        )
        .execute()
    )
    events = events_result.get("items", [])

    if not events:
        # print('No events found.')
        return
    return events

def get_google_service(service: str):
    # Scopes required to access and modify calendar events
    SCOPES = [f"https://www.googleapis.com/auth/{service}"]
    token_local_path = os.path.join(os.environ["TELE_SRC_DIR"], f"{service}_token.json")
    credentials_local_path = os.path.join(
        os.environ["TELE_SRC_DIR"], f"credentials.json"
    )

    # Load credentials from the file downloaded earlier
    creds = None
    if os.path.exists(token_local_path):
        creds = Credentials.from_authorized_user_file(token_local_path, SCOPES)
    # If there are no (valid) credentials available, let the user log in.
    if not creds or not creds.valid:
        if creds and creds.expired and creds.refresh_token:
            creds.refresh(Request())
        else:
            flow = InstalledAppFlow.from_client_secrets_file(
                credentials_local_path, SCOPES
            )
            creds = flow.run_local_server(port=0)
        with open(token_local_path, "w") as token:
            token.write(creds.to_json())

    # Build the service
    service = build(service, "v3", credentials=creds)
    return service

def create_calendar_event(
    calendar_id: str,
    event_name: str,
    event_description: str,
    event_time_start: datetime.datetime,
    event_time_end: datetime.datetime,
    **kwargs,
):
    service = get_calendar_service()

    date_strings = [event_time_start.isoformat(), event_time_end.isoformat()]
    # Define the event details
    event = {
        "summary": event_name,
        "location": "Event Location",
        "description": event_description,
        "start": {
            "dateTime": date_strings[0],  # Start time (in ISO format)
            "timeZone": "UTC",
        },
        "end": {
            "dateTime": date_strings[1],  # End time (in ISO format)
            "timeZone": "UTC",
        },
        **kwargs,
    }

    # Insert the event
    event = service.events().insert(calendarId=calendar_id, body=event).execute()
    print("Event created: %s" % (event.get("htmlLink")))

def num_overlapping_events(
    start_time: datetime.datetime, end_time: datetime.datetime
) -> int:
    """Returns the number of overlapping events in the telescope calendar

    :param start_time: The start time to check
    :type start_time: datetime.datetime
    :param end_time: The end time to check
    :type end_time: datetime.datetime
    :return: The number of overlapping events
    :rtype: int
    """
    events = get_calendar_events_in_range(
        os.environ["POGS_CALENDAR_ID"],
        start_time - datetime.timedelta(minutes=1),
        end_time + datetime.timedelta(minutes=1),
    )
    return 0 if events is None else len(events)

def request_pogs_reservation(
    start_time: datetime.datetime,
    end_time: datetime.datetime,
    event_name: str,
    event_description: str,
):
    assert start_time.tzinfo == datetime.timezone.utc, "start_time must be in UTC"
    assert end_time.tzinfo == datetime.timezone.utc, "end_time must be in UTC"
    assert start_time < end_time, "start_time must be before end_time"

    n_overlap = num_overlapping_events(start_time, end_time)

    if n_overlap > 0:
        raise ValueError(
            f"There are {n_overlap} overlapping events, reservation aborted!"
        )
    else:
        print("No overlapping events found, reservation is valid")
        create_calendar_event(
            os.environ["POGS_CALENDAR_ID"],
            event_name,
            event_description,
            start_time,
            end_time,
        )

def upload_file(
    filename: str,
    filepath: str,
    mimetype: str,
    parent_folder_id: str = None,
    chunksize: int = 5 * 1024 * 1024,
) -> str:
    """Uploads a file to Google Drive

    :param filename: The name of the file when it lands in Google Drive
    :type filename: str
    :param filepath: The path to the file on the local machine
    :type filepath: str
    :param mimetype: The mimetype of the file
    :type mimetype: str
    :param parent_folder_id: The ID of the parent folder, defaults to None
    :type parent_folder_id: str, optional
    :param chunksize: The chunksize to use for uploading, defaults to 10 MB
    :type chunksize: int, optional
    :return: The ID of the file in Google Drive
    :rtype: str
    """
    service = get_drive_service()
    file_metadata = {
        "name": filename,
        "parents": [parent_folder_id] if parent_folder_id else [],
    }
    media = MediaFileUpload(
        filepath, mimetype=mimetype, resumable=True, chunksize=chunksize
    )
    request = service.files().create(body=file_metadata, media_body=media, fields="id")

    response = None
    while response is None:
        # mr.tic()
        status, response = request.next_chunk()
        if status:
            print(f"Uploaded {int(status.progress() * 100)}%")
        # dt = mr.toc(return_elapsed_seconds=True)
        # print(f'Upload speed: {chunksize/dt/1024/1024} MB/s')
    print("Upload Complete! File ID:", response.get("id"))

    return response.get("id")

def create_share_link(file_id: str) -> str:
    """Creates a shareable link for a file in Google Drive

    :param file_id: The ID of the file in Google Drive
    :type file_id: str
    :return: The shareable link
    :rtype: str
    """
    service = get_drive_service()
    service.permissions().create(
        fileId=file_id, body={"role": "reader", "type": "anyone"}
    ).execute()
    return f"https://drive.google.com/file/d/{file_id}/view"

def create_drive_folder(folder_name: str, parent_folder_id: str = None) -> str:
    """Creates a folder in Google Drive

    :param folder_name: The name of the folder to create
    :type folder_name: str
    :param parent_folder_id: The ID of the parent folder, defaults to None
    :type parent_folder_id: str, optional
    :return: The ID of the folder in Google Drive
    :rtype: str
    """
    service = get_drive_service()
    file_metadata = {
        "name": folder_name,
        "mimeType": "application/vnd.google-apps.folder",
        "parents": [parent_folder_id] if parent_folder_id else [],
    }
    file = service.files().create(body=file_metadata, fields="id").execute()
    print(f"Folder '{folder_name}' created with ID: {file.get('id')}")
    return file.get("id")

def zip_and_upload_local_folder(local_path: str, parent_folder_id: str = None) -> str:
    """Zips a local folder and uploads it to Google Drive

    :param local_path: The path to the local folder
    :type local_path: str
    :param parent_folder_id: The ID of the parent folder, defaults to None
    :type parent_folder_id: str, optional
    :return: The link to the zipped folder in Google Drive
    :rtype: str
    """
    zipped_path = os.path.join(os.path.dirname(local_path), "zipped")
    os.makedirs(zipped_path, exist_ok=True)
    zipped_path = os.path.join(zipped_path, os.path.basename(local_path) + ".zip")
    with zipfile.ZipFile(zipped_path, "w") as zipf:
        for root, dirs, files in os.walk(local_path):
            for file in files:
                print(file)
                zipf.write(
                    os.path.join(root, file),
                    os.path.relpath(
                        os.path.join(root, file), os.path.join(local_path, "..")
                    ),
                )
    fid = upload_file(
        os.path.basename(zipped_path),
        zipped_path,
        "application/zip",
        parent_folder_id=parent_folder_id,
    )

    # delete the local zipped folder
    os.remove(zipped_path)
    return create_share_link(fid)
