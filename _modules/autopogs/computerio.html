def size_of_remote_dir(sftp_client: paramiko.SFTPClient, remote_dir: str):
    """Return the size of the remote directory in bytes"""
    total_size = 0
    for filename in sftp_client.listdir(remote_dir):
        if stat.S_ISDIR(sftp_client.stat(remote_dir + filename).st_mode):
            # uses '/' path delimiter for remote server
            total_size += size_of_remote_dir(sftp_client, remote_dir + filename + "/")
        else:
            total_size += sftp_client.stat(remote_dir + filename).st_size
    return total_size

def connect_to_telescope() -> tuple[paramiko.SSHClient, paramiko.SFTPClient]:
    """Connect to the telescope and return the ssh client"""
    ssh = paramiko.SSHClient()
    ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
    ssh.connect(
        hostname=os.environ["POGS_IP"],
        username=os.environ["POGS_USERNAME"],
        password=os.environ["POGS_PASSWORD"],
        port=os.environ["POGS_PORT"],
        look_for_keys=False,
    )
    sftp = ssh.open_sftp()
    return ssh, sftp

def path_exists_remote(sftp_client, path):
    try:
        sftp_client.stat(path)
    except IOError as e:
        if e.errno == errno.ENOENT:
            return False
        raise
    else:
        return True

def download_files(sftp_client, remote_dir, local_dir):
    if not path_exists_remote(sftp_client, remote_dir):
        raise ValueError("Remote path does not exist: " + remote_dir)

    if not os.path.exists(local_dir):
        os.mkdir(local_dir)

    # make sure the remote dir has files in it
    try:
        sftp_client.listdir(remote_dir)
    except IOError:
        raise ValueError("Remote path is a file: " + remote_dir)

    for filename in sftp_client.listdir(remote_dir):
        if stat.S_ISDIR(sftp_client.stat(remote_dir + filename).st_mode):
            # uses '/' path delimiter for remote server
            download_files(
                sftp_client,
                remote_dir + filename + "/",
                os.path.join(local_dir, filename),
            )
        else:
            if not os.path.isfile(os.path.join(local_dir, filename)):
                sftp_client.get(
                    remote_dir + filename, os.path.join(local_dir, filename)
                )

def display_fits_image(fits_path: str, log: bool = True) -> None:
    """Display a fits image"""
    import matplotlib.pyplot as plt
    from astropy.io import fits
    from astropy.visualization import astropy_mpl_style

    plt.style.use(astropy_mpl_style)
    image_data = fits.getdata(fits_path, ext=0)
    data = np.log10(image_data) if log else image_data
    lims = np.percentile(data, [1, 99])
    plt.figure()
    plt.imshow(data, cmap="gray")
    plt.colorbar(label="$\log_{10}(ADU)$" if log else "ADU")
    plt.clim(lims[0], lims[1])
    plt.grid(False)
    plt.show()

    if "TimeKeeper is not running" in out:
        return False
    elif "TimeKeeper is loaded but not active" in out:
        return False
    elif "Primary source offset" in out:
        return True

def write_orchestrate_schedule(
    sftp_client: paramiko.SFTPClient, schedule: list[str]
) -> None:
    """Write the orchestrate schedule to the telescope"""
    # write the .ORC file locally first
    temp_local_schedule = os.path.join(os.environ["TELE_SRC_DIR"], "schedule.txt")
    for line in schedule:
        assert (
            line.count(",") == 2
        ), f'Schedule line "{line}" does not have 2 commas, aborting!'
    with open(temp_local_schedule, "w") as f:
        f.write("\n".join(schedule))
    # copy it to the telescope
    current_schedules = [
        x for x in sftp_client.listdir("D:\\Orchestrate_Watch\\") if x.endswith(".txt")
    ]
    fname = f"{len(current_schedules)+1:08d}.txt"
    sftp_client.put(temp_local_schedule, f"D:\\Orchestrate_Watch\\{fname}")

def download_most_recent_images_tonight(
    sftp_client: paramiko.SFTPClient, local_dir: str, n_images: int = 1
) -> list[str]:
    """Downloads the most recent image taken by the telescope, tonight, returning the local file path"""
    remote_dir = get_tonights_remote_image_directory()
    remote_files = sorted(list_remote_dir(sftp_client, remote_dir))

    if not os.path.exists(local_dir):
        os.mkdir(local_dir)

    local_file_paths = []
    for filename in remote_files[-n_images:]:
        print(f"Downloading {filename}...")
        local_file_path = os.path.join(local_dir, filename)
        get_telescope_file(sftp_client, remote_dir + filename, local_file_path)
        local_file_paths.append(local_file_path)
    return local_file_paths
