def get_current_event():
    """Returns the current event"""
    now = get_now()
    events = get_future_events(os.environ["POGS_CALENDAR_ID"], tol=BUFFER_TIME)
    if events is None:
        return None
    # sort by priority
    events = sorted(
        events, key=lambda x: json.loads(x["description"])["priority"], reverse=True
    )
    for event in events:
        start_date = datetime.datetime.fromisoformat(event["start"]["dateTime"])
        end_date = datetime.datetime.fromisoformat(event["end"]["dateTime"])
        if start_date <= now <= end_date:
            return event
    return None

def cancel_if_moon_angle(event: dict) -> bool:
    """Cancels an event if the moon is too close to the target

    :param event: The event to check
    :type event: dict
    :return: True if the event was cancelled, False otherwise
    :rtype: bool
    """
    desc_dict = json.loads(event["description"])
    obj = mr.SpaceObject("cube.obj", identifier=int(desc_dict["target"]))
    start_date = datetime.datetime.fromisoformat(event["start"]["dateTime"])
    end_date = datetime.datetime.fromisoformat(event["end"]["dateTime"])

    dates = mr.date_linspace(start_date, end_date, 1000)
    r_sat_eci = obj.propagate(dates)
    r_station = station.j2000_at_dates(dates)
    r_hat_eci = mr.hat(r_sat_eci - r_station)

    r_moon = mr.moon(dates)
    r_hat_moon = mr.hat(r_moon - r_station)

    angle_obs_to_moon_rad = mr.angle_between_vecs(r_hat_eci, r_hat_moon)
    min_obs_angle_deg = np.rad2deg(angle_obs_to_moon_rad.min())

    if min_obs_angle_deg < STATION_MOON_EXCLUSION_DEG:
        cancel_event_by_id(
            event["id"],
            f"TARGET IN MOON EXCLUSION ZONE ({min_obs_angle_deg:.2f} < {STATION_MOON_EXCLUSION_DEG} deg",
        )
        return True
    return False

def is_astronomical_night(station: mr.Station, date: datetime.datetime) -> bool:
    """Returns true if the provided datetime is during astronomical night"""
    return np.rad2deg(mr.sun_angle_to_horizon(date, station.itrf)) < -18.0

def get_future_events(
    calendar_id: str, tol: datetime.timedelta = datetime.timedelta(days=365)
) -> list:
    """Returns all future events in the provided calendar"""
    now = get_now()
    events = get_calendar_events_in_range(calendar_id, now, now + tol)
    return events

def cancel_event_by_id(event_id: str, reason: str, silent: bool = False):
    """Cancels the event with the provided ID

    :param event_id: The ID of the event to cancel
    :type event_id: str
    :param reason: The reason for the cancellation
    :type reason: str
    :param silent: If true, do not send cancellation emails, defaults to False
    :type silent: bool, optional
    """
    service = get_calendar_service()
    # first change the summary to prepend "CANCELLED:"
    event = (
        service.events()
        .get(calendarId=os.environ["POGS_CALENDAR_ID"], eventId=event_id)
        .execute()
    )
    event["summary"] = f'{reason}: {event["summary"]}'
    service.events().update(
        calendarId=os.environ["POGS_CALENDAR_ID"], eventId=event_id, body=event
    ).execute()
    print(f'Cancelling event due to "{reason}"')
    service.events().delete(
        calendarId=os.environ["POGS_CALENDAR_ID"],
        eventId=event_id,
        sendUpdates="all" if not silent else "none",
    ).execute()

def cancel_if_day(event: dict) -> bool:
    """Cancels the event if it is during the day

    :param event: The event to check
    :type event: dict
    :return: True if the event was cancelled, False otherwise
    :rtype: bool
    """
    start_date = datetime.datetime.fromisoformat(event["start"]["dateTime"])
    if not is_astronomical_night(station, start_date):
        cancel_event_by_id(event["id"], "NOT ASTRONOMICAL NIGHT")
        return True
    return False

def cancel_if_no_sat_pass(event: dict) -> bool:
    """Cancels an event if the target satellite is not passing overhead at the time

    :param event: The event to check
    :type event: dict
    :return: True if the event was cancelled, False otherwise
    :rtype: bool
    """
    desc_dict = json.loads(event["description"])
    obj = mr.SpaceObject("cube.obj", identifier=int(desc_dict["target"]))
    start_date = datetime.datetime.fromisoformat(event["start"]["dateTime"])
    end_date = datetime.datetime.fromisoformat(event["end"]["dateTime"])

    dates = mr.date_linspace(start_date, end_date, 1000)
    r_sat_eci = obj.propagate(dates)
    r_station = station.j2000_at_dates(dates)
    r_hat_eci = mr.hat(r_sat_eci - r_station)

    az, el = station.eci_to_az_el(dates, r_hat_eci)

    if np.rad2deg(el.max()) < STATION_MIN_ELEVATION_DEG:
        cancel_event_by_id(event["id"], "SATELLITE NEVER ABOVE MINIMUM ELEVATION")
        return True
    return False

def delete_future_events(silent: bool = False):
    """Deletes all future events in the POGS calendar

    :param silent: If true, do not send cancellation emails, defaults to False
    :type silent: bool, optional
    """
    events = get_future_events(os.environ["POGS_CALENDAR_ID"])
    if events is None:
        return
    for event in events:
        cancel_event_by_id(event["id"], "DELETED", silent=silent)

def cancel_if_overlap(event: dict) -> bool:
    """Cancels an event if it overlaps with another event of equal or higher priority

    :param event: The event to check
    :type event: dict
    :return: True if the event was cancelled, False otherwise
    :rtype: bool
    """
    event_start_time = datetime.datetime.fromisoformat(event["start"]["dateTime"])
    event_end_time = datetime.datetime.fromisoformat(event["end"]["dateTime"])
    event_priority = json.loads(event["description"])["priority"]
    existing_events = get_calendar_events_in_range(
        os.environ["POGS_CALENDAR_ID"],
        event_start_time - BUFFER_TIME,
        event_end_time + BUFFER_TIME,
    )
    if existing_events is None:
        return False
    for e_event in existing_events:
        if e_event["id"] == event["id"]:
            continue  # make sure we dont compare the event to itself
        e_event_priority = json.loads(e_event["description"])["priority"]
        if e_event_priority >= event_priority:
            cancel_event_by_id(event["id"], "OVERLAPS WITH HIGHER PRIORITY EVENT")
            return True

def validate_future_events():
    """Validates all future events in the POGS calendar"""
    events = get_future_events(os.environ["POGS_CALENDAR_ID"])
    if events is None:
        return
    for event in events:
        if "VALIDATED" not in event["summary"]:
            was_cancelled = cancel_if_overlap(event)
            if not was_cancelled:
                was_cancelled = cancel_if_day(event)
                if not was_cancelled:
                    was_cancelled = cancel_if_no_sat_pass(event)
                    if not was_cancelled:
                        was_cancelled = cancel_if_moon_angle(event)
                        if not was_cancelled:
                            update_calendar_event_summary(event["id"], "VALIDATED")
