[docs]def rand_cone_vectors2(
    v_base: np.ndarray, angle_tol_deg: float, num: int
) -> np.ndarray:
    if v_base.size == 3:
        v_base = hat(np.tile(v_base, (num, 1)))
    elif v_base.size == num * 3:
        v_base = hat(v_base)
    else:
        raise ValueError(
            "v_base must be either a single 3-vector or a [num x 3] array"
        )
    v = rand_unit_vectors(num)
    while not all(acosd(dot(v_base, v)) < angle_tol_deg):
        failed = (acosd(dot(v_base, v)) > angle_tol_deg).flatten()
        nfailed = np.sum(failed)
        v[failed, :] = rand_unit_vectors(nfailed)
    return v




[docs]def random_camera_params(
    fov_deg: float,
    vmag_max: float,
    cam_r_lims: Tuple[float, float],
    data_points: int,
) -> Tuple[np.ndarray, np.ndarray]:
    camera_pos = rand_point_in_shell(cam_r_lims, data_points)
    camera_rmag = vecnorm(camera_pos)
    camera_look_rmag_max = camera_rmag * tand(fov_deg / 2) - vmag_max
    camera_look_point = rand_point_in_ball(
        camera_look_rmag_max, data_points
    )
    return camera_pos, camera_look_point




[docs]def order_key_points(key_points: np.ndarray) -> np.ndarray:
    nv = key_points.shape[0]
    v_order = [0]
    while len(v_order) < nv:
        v_closest = np.argsort(
            (
                vecnorm(
                    key_points
                    - np.tile(key_points[v_order[-1], :], (nv, 1))
                )
            ).flatten()
        )
        v_next = next(v for v in v_closest if v not in v_order)
        v_order.append(v_next)
    key_points = key_points[v_order, :]
    return key_points




[docs]def key_model_points(
    obj: SpaceObject,
    obj_name: str,
    manual_key_point_selection: bool,
    num_target: int = 30,
) -> np.ndarray:
    dist_tol = 0.5
    key_points = np.array([[]])
    while key_points.shape[0] < num_target:
        dist_tol -= 0.01
        if dist_tol < 0:
            break
        vind_convhull = scipy.spatial.ConvexHull(obj.v).simplices
        uvind_convhull = np.unique(vind_convhull).flatten()
        key_points = obj.v[uvind_convhull, :]
        for _ in range(3):
            for i, v in enumerate(key_points):
                d = vecnorm(
                    np.tile(v, (key_points.shape[0], 1)) - key_points
                ).flatten()
                d[i] = np.nan
                key_points[d < dist_tol, :] = np.nan
            np.delete(key_points, key_points[:, 0] == np.nan, axis=0)

        key_points = key_points[~np.isnan(key_points).any(axis=1)]
        key_points = order_key_points(key_points)

    if manual_key_point_selection:
        scatter_kp = lambda kp: scatter3(
            pl,
            kp,
            point_size=20,
            color="r" if kp.shape[0] < num_target else "g",
            name="kp",
        )

        def callback(pick_point) -> None:
            nonlocal key_points
            match_inds = np.argwhere(
                np.linalg.norm(
                    key_points - pick_point.reshape((-1, 3)), axis=1
                )
                < 1e-2
            ).flatten()
            if match_inds.size:
                print(f"Picked point already a key point, removing...")
                key_points = np.delete(
                    key_points, match_inds[0], axis=0
                )
            else:
                key_points = np.vstack(
                    (key_points, pick_point.reshape((-1, 3)))
                )
                print(
                    f"Added a key point, now we have: {key_points.shape[0]} key points!"
                )
            scatter_kp(key_points)

        print(f"Found {key_points.shape[0]} key points!")
        pl = pv.Plotter()
        obj.render(pl, opacity=0.7, color="gray", pickable=True)
        scatter_kp(key_points)
        pl.enable_point_picking(
            callback=callback,
            pickable_window=False,
            show_point=False,
            left_clicking=True,
            tolerance=0.01,
        )
        pl.show()

    return order_key_points(key_points)




[docs]def intersects_obj(
    origin: np.ndarray, stop: np.ndarray, obj: SpaceObject
) -> bool:
    points, rays, _ = obj._mesh.multi_ray_trace(origin, stop)
    d_int = vecnorm(origin[rays, :] - points)
    rays = np.delete(rays, d_int.flatten() < 1e-1)
    vis = np.ones((origin.shape[0],), dtype=bool)
    vis[np.unique(rays)] = False
    return vis




[docs]def key_point_visibility(
    model_key_points: np.ndarray,
    sun_dir: np.ndarray,
    camera_pos: np.ndarray,
    camera_look_point: np.ndarray,
    obj: SpaceObject,
    kp_pix_vals: np.ndarray,
) -> np.ndarray:
    nkp = model_key_points.shape[0]
    vec_to_sun = 10 * np.tile(hat(sun_dir), (nkp, 1))
    sun_stop = model_key_points + vec_to_sun
    is_illum = intersects_obj(model_key_points, sun_stop, obj)
    is_observ = intersects_obj(
        model_key_points, np.tile(camera_pos, (nkp, 1)), obj
    )
    return np.logical_and(
        np.logical_and(is_illum, is_observ), kp_pix_vals > 0
    )




[docs]def camera_rmag_limits(obj: SpaceObject) -> np.ndarray:
    mdisp = np.max(vecnorm(obj.v))
    return mdisp * np.array([6, 10])




[docs]def model_parameters(
    obj_name: str,
    obj: SpaceObject,
    key_point_count: int,
    manual_key_point_selection: bool,
) -> Tuple[np.ndarray, Tuple]:
    model_key_points = key_model_points(
        obj,
        obj_name,
        num_target=key_point_count,
        manual_key_point_selection=manual_key_point_selection,
    )
    cam_r_lims = camera_rmag_limits(obj)

    return (model_key_points, cam_r_lims)

[docs]def generate_synthetic_dataset(
    obj_name: str,
    data_points: int = 1_000,
    sz: int = 512,
    fov_deg: float = 25.0,
    max_phase_deg: float = 90.0,
    key_point_count: int = 30,
    clear_out_dir: bool = True,
    manual_key_point_selection: bool = True,
) -> None:
    if clear_out_dir:
        os.system("rm -rf out")
        os.mkdir("out")
    obj = SpaceObject(obj_name)
    (model_key_points, cam_r_lims) = model_parameters(
        obj_name, obj, key_point_count, manual_key_point_selection
    )
    v_max = max(vecnorm(obj.v))
    brdf_ind = 1
    camera_pos, camera_look_point = random_camera_params(
        fov_deg, v_max, cam_r_lims, data_points
    )
    camera_up = rand_unit_vectors(data_points)
    sun_vectors = (
        5
        * v_max
        * rand_cone_vectors2(camera_pos, max_phase_deg, data_points)
    )

    cam_near = 0.01
    cam_far = cam_r_lims[1] + v_max

    run(
        sun_vectors,
        camera_pos,
        camera_look_point,
        camera_up,
        sz,
        obj_name,
        brdf_ind,
        cam_far,
        fov_deg,
    )

    camera_projections = np.empty((4, 4, data_points))
    camera_views = np.empty((4, 4, data_points))
    model_view_projection_matrices = np.empty((4, 4, data_points))
    kp_im_coords = []
    for i in range(data_points):
        V = look_at_matrix(
            camera_pos[i], camera_look_point[i], camera_up[i]
        )
        P = perspective_projection_matrix(
            fov_deg=fov_deg, aspect=1, near=cam_near, far=cam_far
        )
        mvp = P @ V @ np.identity(4)
        camera_projections[:, :, i] = P
        camera_views[:, :, i] = V
        model_view_projection_matrices[:, :, i] = mvp

        kp_homo_coords = []
        for kp in model_key_points:
            kp_homo_coords.append(
                mvp @ np.reshape(np.concatenate((kp, [1])), (4, 1))
            )
        kp_homo_coords = np.hstack(kp_homo_coords).T
        kp_homo_coords = kp_homo_coords[:, :3] / kp_homo_coords[:, [3]]
        kp_im_coords.append(
            (kp_homo_coords[:, :2] * sz / 2 + sz / 2).astype(np.int32)
        )

    is_vis = []
    for i in range(data_points):
        tic()
        img = np.asarray(Image.open(f"out/frame{i}.png"))
        kp_pix_vals = img[
            kp_im_coords[i][:, 1], kp_im_coords[i][:, 0], 0
        ]
        is_vis.append(
            key_point_visibility(
                model_key_points,
                sun_vectors[i],
                camera_pos[i],
                camera_look_point[i],
                obj,
                kp_pix_vals,
            )
            + 1
        )
        dt = toc(return_elapsed_seconds=True)
        print(
            f"Computed visibility for point {i+1}/{data_points}, {(data_points - i - 1) * dt/60:.2f}ish minutes remaining"
        )

    quats = dcm_to_quat(camera_views[:3, :3, :])
    translations = camera_views[:3, 3, :].T

    iminfo = dict()
    iminfo["key_points"] = {
        "sat_body_frame": model_key_points,
        "image_pixels": kp_im_coords,
        "visibility": is_vis,
    }
    iminfo["fov"] = fov_deg
    iminfo["phase_angle"] = acosd(
        dot(hat(camera_pos), hat(sun_vectors))
    )
    iminfo["sun_model_frame"] = sun_vectors
    iminfo["camera_pos_model_frame"] = camera_pos
    iminfo["camera_up_model_frame"] = camera_up
    iminfo["camera_look_point_model_frame"] = camera_look_point
    iminfo["body_to_camera_quats"] = quats
    iminfo["model_translation_camera_frame"] = translations
    iminfo["camera_projection_matrices"] = camera_projections
    iminfo["camera_view_matrices"] = camera_views
    iminfo[
        "model_view_projection_matrices"
    ] = model_view_projection_matrices
    iminfo["camera_far"] = cam_far
    iminfo["camera_near"] = cam_near
    iminfo["image_dims"] = sz
    iminfo["file_names"] = [
        f"frame_{i}.png" for i in range(data_points)
    ]
    iminfo["image_count"] = data_points

    iminfo["brdf"] = "phong"
    iminfo["obj"] = {
        "vertices": obj.v,
        "faces": obj.f,
        "obj_name_name": obj_name,
        "mtl_file_name": "tess.mtl",
    }

    json_dump = json.dumps(iminfo, cls=NumpyEncoder)
    with open("out/iminfo.json", "w") as f:
        f.write(json_dump)
