    def _f_engine(
        self,
        laz: np.ndarray,
        lel: np.ndarray,
        oaz: np.ndarray,
        oel: np.ndarray,
    ) -> np.ndarray:
        """True brightness function, evaluates using the LightCurveEngine renderer

        :param laz: Lighting azimuth in the body frame [rad]
        :type laz: np.ndarray
        :param lel: Lighting elevation in the body frame [rad]
        :type lel: np.ndarray
        :param oaz: Observer azimuth in the body frame [rad]
        :type oaz: np.ndarray
        :param oel: Observer elevation in the body frame [rad]
        :type oel: np.ndarray
        :return: Unit distance irradiance in given observation geometry [W/m^2]
        :rtype: np.ndarray
        """
        (x1, y1, z1) = sph_to_cart(laz, lel, lel * 0 + 1)
        L = np.vstack((x1, y1, z1)).T
        (x2, y2, z2) = sph_to_cart(oaz, oel, oel * 0 + 1)
        O = np.vstack((x2, y2, z2)).T
        if self.use_engine:
            nper = 1e4
            runcount = int(np.ceil(L.shape[0] / nper))
            lc = []
            for i in range(runcount):
                lower_ind = int(i * nper)
                upper_ind = min(int((i + 1) * nper), L.shape[0])
                inds = list(range(lower_ind, upper_ind))
                lc.append(
                    run_engine(
                        self.brdf,
                        self.model_name,
                        L[inds, :],
                        O[inds, :],
                    )
                )
            return np.vstack(lc)
        else:
            obj = SpaceObject(self.model_name)
            lc = obj.compute_convex_light_curve(self.brdf, L, O)
            return np.reshape(lc, (lc.size, 1))

        self.file_name = f"{self.file_without_ext}.pkl"


[docs]    def build_from_engine(self, dimn: int = 10) -> None:
        """Builds the interpolated model, enables ``self.eval(points)`` after completion

        :param dimn: Size of each linear dimension, defaults to 10
        :type dimn: int, optional
        """
        assert dimn > 0 and isinstance(
            dimn, int
        ), "Model order must be a positive integer"
        self.order = dimn
        saz, sel = np.linspace(-np.pi, np.pi, dimn), np.linspace(
            -np.pi / 2, np.pi / 2, dimn
        )
        self.grid_pts = (saz, sel, saz, sel)
        (gaz1, gel1, gaz2, gel2) = np.meshgrid(
            saz, sel, saz, sel, indexing="ij"
        )
        lc = self._f_engine(
            gaz1.flatten(),
            gel1.flatten(),
            gaz2.flatten(),
            gel2.flatten(),
        )
        self.data = np.reshape(lc, gaz1.shape)
        self.make_interpolator()

[docs]    def load_from_file(self, path: str) -> None:
        """Loads and builds the interpolator from a saved pickle file containing the brightness data and interpolator grids

        :param path: Path to the pickle file containing the saved data
        :type path: str
        """
        with open(path, "rb") as f:
            d = pickle.load(f)
        self.data = np.array(d["data"])
        self.grid_pts = tuple([np.array(g) for g in d["grid_pts"]])
        self.brdf = Brdf(
            name=d["brdf"]["name"],
            cd=d["brdf"]["cd"],
            cs=d["brdf"]["cs"],
            n=d["brdf"]["n"],
        )
        self.model_name = d["model_name"]
        self.order = self.grid_pts[0].shape[0]
        self.make_interpolator()



[docs]    def make_interpolator(self) -> None:
        """Builds the interpolator"""
        self.interp = scipy.interpolate.RegularGridInterpolator(
            self.grid_pts, self.data, method=self.method
        )
        self.is_built = True



[docs]    def eval(self, L: np.ndarray, O: np.ndarray) -> np.ndarray:
        """Evaluates the brightness model with the given observer and sun unit vectors in the body frame

        :param L: Sun vector in body frame, length irrelevant
        :type L: np.ndarray [nx3]
        :param O: Observer vector in body frame, length irrelevant
        :type O: np.ndarray [nx3]
        :raises AttributeError: If the model has not been built yet
        :return: Interpolated normalized irradiances [W/m^2]
        :rtype: np.ndarray [nx1]
        """
        if self.is_built:
            (az1, el1, _) = cart_to_sph(L[:, 0], L[:, 1], L[:, 2])
            (az2, el2, _) = cart_to_sph(O[:, 0], O[:, 1], O[:, 2])
            return self.interp((az1, el1, az2, el2))
        else:
            raise AttributeError("Interpolator not available")

    """
    Predicts brightness by evaluating full non-convex rendered brightness over a 6D cloud of lighting and illumination unit vectors, used to train a Multi-Layer Perceptron (MLP)

    """

    def __init__(
        self,
        obj: SpaceObject = None,
        brdf: Brdf = None,
        path: str = None,
        use_engine: bool = True,
        layers: tuple[int] = (150, 50, 150),
        train_on: str = "irradiance",
    ):
        assert (
            train_on == "irradiance" or train_on == "magnitude"
        ), "train_on must be magnitude or irradiance"
        self.train_on = train_on
        self.layers = layers
        self.use_engine = use_engine
        if obj is not None and brdf is not None:
            self.brdf = brdf
            self.obj = obj
        elif path is not None:
            self.load_from_file(path)
        self.file_without_extension = f"mlp_model_train_on_{self.train_on}_{self.obj.file_name[:-4]}_brdf_{self.brdf.name}_cd_{self.brdf.cd}_cs_{self.brdf.cs}_n_{self.brdf.n}_layers_{'_'.join([str(x) for x in self.layers])}"
        self.sklearn_file_name = f"{self.file_without_extension}.pkl"
        self.onnx_file_name = f"{self.file_without_extension}.onnx"


[docs]    def brightness(
        self, svb: np.ndarray, ovb: np.ndarray, output_as: str = None
    ) -> np.ndarray:
        """True brightness function, evaluates using the ``LightCurveEngine`` renderer or a convex BRDF method

        :param svb: Sun unit vector in the body frame
        :type svb: np.ndarray [nx3]
        :param ovb: Observer unit vectors in the body frame
        :type ovb: np.ndarray [nx3]
        :param output_as: Values to return, 'irradiance' or 'magnitude', defaults to None (uses ``self.train``)
        :type output_as: str
        :return: Unit distance irradiance in given observation geometry [W/m^2]
        :rtype: np.ndarray
        """
        if self.use_engine:
            nper = 1e4
            runcount = int(np.ceil(svb.shape[0] / nper))
            lc = []
            for i in range(runcount):
                lower_ind = int(i * nper)
                upper_ind = min(int((i + 1) * nper), svb.shape[0])
                inds = list(range(lower_ind, upper_ind))
                lc.append(
                    run_engine(
                        self.brdf,
                        self.obj.file_name,
                        svb[inds, :],
                        ovb[inds, :],
                    )
                )
            b = np.vstack(lc)
        else:
            b = self.obj.compute_convex_light_curve(self.brdf, svb, ovb)
        b[b == 0] = 1e-5
        output_as = self.train_on if output_as is None else output_as
        return (
            b
            if output_as == "irradiance"
            else irradiance_to_apparent_magnitude(b)
        )



[docs]    def gen_train_dataset(
        self, num: int
    ) -> Tuple[np.ndarray, np.ndarray]:
        num = int(np.ceil(np.sqrt(num)))
        pts = fibonacci_sample_sphere(num)
        (si, vi) = np.meshgrid(
            np.arange(num), np.arange(num), indexing="ij"
        )
        svb = pts[si.flatten()]
        ovb = pts[vi.flatten()]
        X_train = np.hstack((svb, ovb))
        tic("Compute training LC")
        lc_train = self.brightness(svb, ovb)
        toc()
        return (X_train, lc_train)



[docs]    def train(self, num_train: int = 100_000) -> None:
        """Builds the MLP model, enables ``self.eval(points)`` after completion

        :param num_train: Number of points to train on, actually uses closest square
        :type num_train: int, optional
        """
        assert num_train > 0 and isinstance(
            num_train, int
        ), "num_train must be a positive integer"

        mlp = MLPRegressor(
            hidden_layer_sizes=self.layers,
            activation="relu",
            solver="adam",
            verbose=True,
            learning_rate="adaptive",
            tol=1e-4,
        )
        self.model = make_pipeline(PowerTransformer(), mlp)
        (X_train, lc_train) = self.gen_train_dataset(num_train)
        tic(f"Fit against {num_train} pts: ")
        self.model.fit(X_train, lc_train)
        toc()
        tic("Serialize model")
        self.onx = to_onnx(self.model, X_train[:1].astype(np.float32))
        self.sess = InferenceSession(self.onx.SerializeToString())
        toc()



[docs]    def save_to_file(
        self, directory: str = "", save_as_format: str = "onnx"
    ) -> None:
        """Saves the model to the directory provided

        :param directory: Directory to save model in, defaults to ""
        :type directory: str, optional
        :param save_as_format: Format to save the file to, either 'onnx' or 'sklearn', defaults to 'onnx'

        if save_as_format == "sklearn":

                pickle.dump(self.model, f)
        elif save_as_format == "onnx":
            save_file_path = os.path.join(
                directory, self.onnx_file_name
            )
            with open(save_file_path, "wb") as f:
                f.write(self.onx.SerializeToString())
        else:
            raise NotImplementedError(
                "save_as_format must be 'onnx' or 'sklearn'"
            )

[docs]    def eval(
        self,
        svb: np.ndarray,
        ovb: np.ndarray,
        eval_mode_pref: str = "sklearn",
        output_as: str = None,
    ) -> np.ndarray:
        """Evaluates the brightness model with the given observer and sun unit vectors in the body frame

        :param svb: Sun vector in body frame, length irrelevant
        :type svb: np.ndarray [nx3]
        :param ovb: Observer vector in body frame, length irrelevant
        :type ovb: np.ndarray [nx3]
        :param eval_mode_pref: Evaluation mode preference, defaults to 'sklearn'
        :type eval_mode_pref: str
        :param output_as: Values to return, 'irradiance' or 'magnitude', defaults to None (``self.train_on`` is used)
        :type output_as: str
        :raises AttributeError: If the model has not been built yet
        :return: Predicted normalized irradiances [W/m^2]
        :rtype: np.ndarray [nx1]
        """
        assert eval_mode_pref in [
            "sklearn",
            "onnx",
        ], "eval_mode_pref must be 'sklearn' or 'onnx'"
        if not hasattr(self, "model") and not hasattr(self, "sess"):
            raise AttributeError(
                "sklearn or onnx models not available, load or train one first"
            )
        x = np.hstack((ovb, svb))
        if eval_mode_pref == "sklearn":
            if hasattr(self, "model"):
                b = self.model.predict(x)
            elif hasattr(self, "sess"):
                b = self.sess.run(None, {"X": x.astype(np.float32)})[
                    0
                ].flatten()
        elif eval_mode_pref == "onnx":
            if hasattr(self, "sess"):
                b = self.sess.run(None, {"X": x.astype(np.float32)})[
                    0
                ].flatten()
            elif hasattr(self, "model"):
                b = self.model.predict(x)

        output_as = self.train_on if output_as is None else output_as
        if output_as == "irradiance":
            return (
                b
                if self.train_on == "irradiance"
                else apparent_magnitude_to_irradiance(b)
            )
        elif output_as == "magnitude":
            return (
                irradiance_to_apparent_magnitude(b)
                if self.train_on == "irradiance"
                else b
            )
        else:
            raise ValueError(
                "output_as must be 'irradiance' or 'magnitude'"
            )
